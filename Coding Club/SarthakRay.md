```C++
// C++ Code which gives SUM of ASCII values of your FULL NAME

#include <stdio.h>
#include <iostream>

int main()
{
    char Name[] = {'S','A','R','T','H','A','K',' ','R','A','Y','\0'};
    int sum = 0;

    for(int i=0; i<sizeof(Name)/sizeof(Name[0]); i++)
        {
            sum += int(Name[i]);
        }

    std::cout << sum << std::endl;    

    return 0;
}
```
